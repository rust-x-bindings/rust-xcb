// types

#[derive(Debug, Clone, Copy, PartialEq)]
#[repr(C)]
pub struct XineramaScreenInfo {
    pub screen_number: c_int,
    pub x_org: c_short,
    pub y_org: c_short,
    pub width: c_short,
    pub height: c_short,
}

#[derive(Debug, Clone, Copy, PartialEq)]
#[repr(C)]
pub struct XPanoramiXInfo {
    pub window: Window,
    pub screen: c_int,
    pub State: c_int,
    pub width: c_int,
    pub height: c_int,
    pub ScreenCount: c_int,
    pub eventMask: XID,
}
