67    /**
68     * @brief Send a request to the server.
69     * @param c The connection to the X server.
70     * @param flags A combination of flags from the xcb_send_request_flags_t enumeration.
71     * @param vector Data to send; must have two iovecs before start for internal use.
72     * @param request Information about the request to be sent.
73     * @param num_fds Number of additional file descriptors to send to the server
74     * @param fds Additional file descriptors that should be send to the server.
75     * @return The request's sequence number on success, 0 otherwise.
76     *
77     * This function sends a new request to the X server. The data of the request is
78     * given as an array of @c iovecs in the @p vector argument. The length of that
79     * array and the necessary management information are given in the @p request
80     * argument.
81     *
82     * If @p num_fds is non-zero, @p fds points to an array of file descriptors that
83     * will be sent to the X server along with this request. After this function
84     * returns, all file descriptors sent are owned by xcb and will be closed
85     * eventually.
86     *
87     * When this function returns, the request might or might not be sent already.
88     * Use xcb_flush() to make sure that it really was sent.
89     *
90     * Please note that this function is not the preferred way for sending requests.
91     *
92     * Please note that xcb might use index -1 and -2 of the @p vector array internally,
93     * so they must be valid!
94     */
95    pub(crate) fn xcb_send_request_with_fds(

149    /**
150     * @brief Send a file descriptor to the server in the next call to xcb_send_request.
151     * @param c The connection to the X server.
152     * @param fd The file descriptor to send.
153     *
154     * After this function returns, the file descriptor given is owned by xcb and

157     * @deprecated This function cannot be used in a thread-safe way. Two threads

163    /**
164     * @brief Take over the write side of the socket
165     * @param c The connection to the X server.
166     * @param return_socket Callback function that will be called when xcb wants
167     *                        to use the socket again.
168     * @param closure Argument to the callback function.
169     * @param flags A combination of flags from the xcb_send_request_flags_t enumeration.
170     * @param sent Location to the sequence number of the last sequence request.
171     *              Must not be NULL.
172     * @return 1 on success, else 0.
173     *
174     * xcb_take_socket allows external code to ask XCB for permission to
175     * take over the write side of the socket and send raw data with
176     * xcb_writev. xcb_take_socket provides the sequence number of the last
177     * request XCB sent. The caller of xcb_take_socket must supply a
178     * callback which XCB can call when it wants the write side of the
179     * socket back to make a request. This callback synchronizes with the
180     * external socket owner and flushes any output queues if appropriate.
181     * If you are sending requests which won't cause a reply, please note the
182     * comment for xcb_writev which explains some sequence number wrap issues.
183     *
184     * All replies that are generated while the socket is owned externally have
185     * @p flags applied to them. For example, use XCB_REQUEST_CHECK if you don't
186     * want errors to go to xcb's normal error handling, but instead having to be
187     * picked up via xcb_wait_for_reply(), xcb_poll_for_reply() or
188     * xcb_request_check().
189     */
190    pub(crate) fn xcb_take_socket(
191        c: *mut xcb_connection_t,
192        return_socket: fn(closure: *mut c_void),
193        closure: *mut c_void,
194        flags: c_int,
195        sent: *mut u64,
196    ) -> c_int;

198    /**
199     * @brief Send raw data to the X server.
200     * @param c The connection to the X server.
201     * @param vector Array of data to be sent.
202     * @param count Number of entries in @p vector.
203     * @param requests Number of requests that are being sent.
204     * @return 1 on success, else 0.
205     *
206     * You must own the write-side of the socket (you've called
207     * xcb_take_socket, and haven't returned from return_socket yet) to call
208     * xcb_writev. Also, the iovec must have at least 1 byte of data in it.
209     * You have to make sure that xcb can detect sequence number wraps correctly.
210     * This means that the first request you send after xcb_take_socket must cause a
211     * reply (e.g. just insert a GetInputFocus request). After every (1 << 16) - 1
212     * requests without a reply, you have to insert a request which will cause a
213     * reply. You can again use GetInputFocus for this. You do not have to wait for
214     * any of the GetInputFocus replies, but can instead handle them via
215     * xcb_discard_reply().
216     */
217    pub(crate) fn xcb_writev(
218        c: *mut xcb_connection_t,
219        vector: *mut iovec,
220        count: c_int,
221        requests: u64,
222    ) -> c_int;
